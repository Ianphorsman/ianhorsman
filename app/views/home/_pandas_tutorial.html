for row in sd.itertuples():
    if row.state in df_2_states:
        state = row.state
        deaths_by_ethnicity = df_2.ix[df_2.state == state].groupby('raceethnicity').size()
        D_T = deaths_by_ethnicity.sum()
        sd.loc[sd.state == state, 'total_deaths'] = D_T
        P_black = row.black
        P_white = row.white
        P_hispanic = row.hispanic
        if 'Black' in deaths_by_ethnicity:
            D_black = deaths_by_ethnicity['Black']
            L_black = (D_black / D_T)
            sd.loc[sd.state == state, 'num_shot_black'] = D_black
            sd.loc[sd.state == state, 'prob_shot_black'] = L_black
        if 'White' in deaths_by_ethnicity:
            D_white = deaths_by_ethnicity['White']
            L_white = (D_white / D_T)
            sd.loc[sd.state == state, 'num_shot_white'] = D_white
            sd.loc[sd.state == state, 'prob_shot_white'] = L_white
        if 'Hispanic/Latino' in deaths_by_ethnicity:
            D_hispanic = deaths_by_ethnicity['Hispanic/Latino']

            sd.loc[sd.state == state, 'num_shot_hispanic'] = D_hispanic
